WhatFizz Buzz is a counting game. Players begin counting at 1, and move around the circle, increasing the count with each player. In every game there are two special numbers, representing "fizz" and "buzz". If the number the player is up to is divisible by this number, the player must say "fizz" or "buzz" respectively. If the number is divisible by both, the player must say "fizz buzz".
For example, if "fizz" is 3 and "buzz" is 5, players would count: 1, 2, Fizz, 4, Buzz, Fizz, 7, 8, Fizz, Buzz, 11, Fizz, 13, 14, Fizz Buzz, 16, 17, Fizz, 19, Buzz, etc.

function fizzBuzz(fizz, buzz, end){
            
  var answer = [];
            
  for (var i = 1; i <= end; i++){
            
    if (i % fizz === 0 && i % buzz === 0){
            
    answer.push("Fizz Buzz");
            
    } else if (i % fizz === 0){
            
    answer.push("Fizz");
            
    } else if (i % buzz === 0){
            
    answer.push("Buzz");
            
    } else {
            
    answer.push(i);
            
    }
            
  }
            
  console.log(answer.join(", "));
            
}
